      remove_idea = CHI_covid_idea_3

    }

  else_if = {

      has_idea = CHI_covid_idea_4
    }
    swap_ideas = {
      remove_idea = CHI_covid_idea_4

PRC_Infiltration_change = {
  custom_effect_tooltip = PRC_Infiltration_change_tooltip
  # 進度累加
  add_to_variable = {
    var = CHI_PRC_Infiltration_progress
    value = CHI_PRC_Infiltration_progress_temp
  }
  clamp_variable = {
    var = CHI_PRC_Infiltration_progress
    min = 0
    max = 100
  }
  # 用 number 當計算用變數
  set_variable = {
    var = number
    value = CHI_PRC_Infiltration_progress_temp
  }
  divide_variable = {
    var = number
    value = 1000
  }
  multiply_variable = {
    var = number
    value = -1
  }
  # 累加 buff
  add_to_variable = {
    var = CHI_PRC_Infiltration_buff
    value = number
  }
  clamp_variable = {
    var = CHI_PRC_Infiltration_buff
    min = -0.1
    max = 0
  }
  if = {
    limit = {
      check_variable = { CHI_PRC_Infiltration_progress < 25 }
    }
    hidden_effect = {
      remove_ideas = CHI_PRC_infiltration_low
      remove_ideas = CHI_PRC_infiltration_middle
      remove_ideas = CHI_PRC_infiltration_high
    }
  }
  else_if = {
    limit = {
      check_variable = { CHI_PRC_Infiltration_progress > 10 }
      check_variable = { CHI_PRC_Infiltration_progress < 50 }
    }
    hidden_effect = {
      add_ideas = CHI_PRC_infiltration_low
      remove_ideas = CHI_PRC_infiltration_middle
      remove_ideas = CHI_PRC_infiltration_high
    }
  }
  else_if = {
    limit = {
      check_variable = { CHI_PRC_Infiltration_progress > 50 }
      check_variable = { CHI_PRC_Infiltration_progress < 99 }
    }
    hidden_effect = {
      remove_ideas = CHI_PRC_infiltration_low
      add_ideas = CHI_PRC_infiltration_middle
      remove_ideas = CHI_PRC_infiltration_high
    }
  }
  else_if = {
    limit = { check_variable = { CHI_PRC_Infiltration_progress = 100 } }
    hidden_effect = {
      remove_ideas = CHI_PRC_infiltration_low
      remove_ideas = CHI_PRC_infiltration_middle
      add_ideas = CHI_PRC_infiltration_high
    }
  }
}

PDTO_unity_increase2 = {
  custom_effect_tooltip = PDTO_unity_tt1
  every_country = {
    limit = {
      OR = {
        has_country_flag = pdto_member
      }
    }
    if = {
      limit = {
        has_country_flag = PDTO_unity
        has_idea = PDTO_unity_1
      }
      swap_ideas = {
        remove_idea = PDTO_unity_1
        add_idea = PDTO_unity_2
      }
    }
    else_if = {
      limit = {
        has_country_flag = PDTO_unity
        has_idea = PDTO_unity_2
      }
      swap_ideas = {
        remove_idea = PDTO_unity_2
        add_idea = PDTO_unity_3
      }
    }
    else_if = {
      limit = {
        has_country_flag = PDTO_unity
        has_idea = PDTO_unity_3
      }
      swap_ideas = {
        remove_idea = PDTO_unity_3
        add_idea = PDTO_unity_4
      }
    }
    else_if = {
      limit = {
        has_country_flag = PDTO_unity
        has_idea = PDTO_unity_4
      }
      swap_ideas = {
        remove_idea = PDTO_unity_4
        add_idea = PDTO_unity_5
      }
    }
    else_if = {
      limit = {
        NOT = {
          has_country_flag = PDTO_unity
          has_ideas = PDTO_unity_1
        }
      }
      set_country_flag = PDTO_unity
      add_ideas = PDTO_unity_1
    }
  }

}
