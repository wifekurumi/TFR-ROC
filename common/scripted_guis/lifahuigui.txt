	TFR_CITS_gui = {
		context_type = player_context
		parent_window_token = politics_tab
		window_name = "CITS_button_container"

		ai_enabled = {
			always = no
		}
	
		visible = {
			has_country_flag = PRC_CITS_members
		}

		triggers = {
			TFR_CITS_button_visible = {
				NOT = {
					
				}
			}
			TFR_CITS_button_click_enabled = {
				NOT = {
					ROOT = { has_country_flag = CITS_list_open }
				}
			}
		}

		effects = {
			TFR_CITS_button_click = {
				add_to_array = { ROOT.CITS_list_array = PRC }
				for_each_loop = {
					array = PRC.PRC_CITS_members
					if = {
						limit = {
							NOT = { is_in_array = { ROOT.CITS_list_array = v } }
						}
						add_to_array = { ROOT.CITS_list_array = v }
					}
				}
				ROOT = {
					set_variable = { CITS_leader = PRC }
					set_country_flag = CITS_list_open
				}
			}
		}
	}
